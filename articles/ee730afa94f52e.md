# はじめに
2月23日にFastAPI 0.131.0がリリースされました。

https://twitter.com/FastAPI/status/2025627146866073881

中でも注目されたのが以下のツイートです。

https://twitter.com/FastAPI/status/2025613852394529114

要約すると、「戻り値に型だけ指定することで、レスポンスの速さが最大2倍になる」とのことです。

え？すごくないすか？

# なんで早くなったん？

以下のPRを読み解きます。

https://github.com/fastapi/fastapi/pull/14962

```
✨ Serialize JSON response with Pydantic (in Rust), when there's a Pydantic return type or response model

訳:Pydantic の戻り値型またはレスポンスモデルがある場合、Pydantic (Rust 内) を使用して JSON レスポンスをシリアル化します
```

めっちゃシンプルな説明ですね。
PydanticはPythonで使われているバリデーションライブラリで、主にAPI周りの入出力の型チェックに使われています。

PydanticでRust?と思われますが、実は現行のPydantic v2はRustで書かれており、v1に比べて倍以上の処理速度を実現しているとのこと。

早くなっただけでなく、v2ではデフォルトでjsonシリアライズに対応するようになり、Pydanticのみでシリアライズが可能になったそうです。
つまりRust側でjsonシリアライズすることで高速になったそうです。

:::details もうちょい補足

大きく変わったのはPRのこの部分。

https://github.com/fastapi/fastapi/blob/880ad19087c65436da083118aad4b4b9f4da1503/fastapi/routing.py#L290-L291

dump_jsonが有効の場合、serialize_jsonを使ってjsonシリアライズをします。これがRustでシリアライズする関数になります。

https://github.com/fastapi/fastapi/blob/880ad19087c65436da083118aad4b4b9f4da1503/fastapi/_compat/v2.py#L202-L226

一方でdump_jsonが無効の場合は従来通りserializeが使われます。

https://github.com/fastapi/fastapi/blob/880ad19087c65436da083118aad4b4b9f4da1503/fastapi/_compat/v2.py#L177-L200

dump_pythonはpythonオブジェクトを返却するため、それをjsonに変換する手間がありました。

Rustのパフォーマンスに加えて、pythonを経由せずにjsonシリアライズすることになり高速になったそうです。
:::

# ベンチマーク
では**見せてもらおうか、新しいPydanticの性能とやらを！**
ということでベンチマークで早くなったのか検証します。

## 方法
以下の2つのエンドポイントを作成し、負荷テストで検証したいと思います。
- 従来の`response_model`を使う方法
- 戻り値の型ヒント（Return Type）を使う方法(New!)

大量のデータを返すシナリオが最も差が出やすいため、1,000個のアイテムを返す設定で比較します。(どんなエンドポイントや)

```python
from fastapi import FastAPI
from pydantic import BaseModel

app = FastAPI()

class User(BaseModel):
    id: int
    name: str
    email: str
    is_active: bool

# ダミーデータ1,000件生成
test_data: list[User] = [
    User(
        id=i,
        name=f"User{i}",
        email=f"user{i}@example.com",
        is_active=i % 2 == 0
    )
    for i in range(1000)
]

@app.get("/old-way", response_model=list[User])
def get_users_old_way():
    """従来の response_model 引数を使用"""
    return test_data

@app.get("/new-way")
def get_users_new_way() -> list[User]:
    """戻り値型ヒントを使用"""
    return test_data

@app.get("/health")
def health_check():
    """ヘルスチェック用"""
    return {"status": "ok"}

```

## 検証
こちらも2種で比較します。
- 計測ツールheyを使って、それぞれに10000回リクエストを送る。
- pythonで計測用スクリプトを書いて、各エンドポイントに対して 100回〜500回 程度のリクエストを送信する。

### 100回〜500回 程度のリクエスト

以下のような計測ツールで検証します。

:::details ベンチマークスクリプト
```python
import asyncio
import time
import statistics
import sys
import httpx
from typing import List


async def benchmark_endpoint(client: httpx.AsyncClient, url: str, requests_count: int) -> dict:
    """エンドポイントのベンチマークを実行"""
    response_times = []
    
    print(f"\n🔄 Testing {url} with {requests_count} requests...")
    
    # ウォームアップ（最初の数回は除外）
    for _ in range(5):
        await client.get(url)
    
    start_time = time.perf_counter()
    
    for i in range(requests_count):
        request_start = time.perf_counter()
        response = await client.get(url)
        request_end = time.perf_counter()
        
        if response.status_code != 200:
            print(f"❌ Error: Status code {response.status_code}")
            continue
            
        response_times.append((request_end - request_start) * 1000)  # ms
        
        if (i + 1) % 100 == 0:
            print(f"  Progress: {i + 1}/{requests_count} requests completed")
    
    total_time = time.perf_counter() - start_time
    
    return {
        "url": url,
        "total_requests": requests_count,
        "total_time": total_time,
        "avg_response_time": statistics.mean(response_times),
        "median_response_time": statistics.median(response_times),
        "min_response_time": min(response_times),
        "max_response_time": max(response_times),
        "rps": requests_count / total_time,
        "stdev": statistics.stdev(response_times) if len(response_times) > 1 else 0
    }


async def run_benchmarks(base_url: str, requests_count: int):
    """すべてのベンチマークを実行"""
    async with httpx.AsyncClient(base_url=base_url, timeout=30.0) as client:
        # ヘルスチェック
        print("🏥 Health check...")
        health_response = await client.get("/health")
        if health_response.status_code != 200:
            print("❌ Server is not healthy!")
            sys.exit(1)
        print("✅ Server is healthy\n")
        
        # ベンチマーク実行
        results = []
        
        # Old way (response_model)
        old_way_result = await benchmark_endpoint(client, "/old-way", requests_count)
        results.append(("Old Way (response_model)", old_way_result))
        
        # 少し待機
        await asyncio.sleep(1)
        
        # New way (return type annotation)
        new_way_result = await benchmark_endpoint(client, "/new-way", requests_count)
        results.append(("New Way (return type)", new_way_result))
        
        return results


def print_report(results: List[tuple], python_version: str, fastapi_version: str, pydantic_version: str):
    """結果レポートを出力"""
    print("\n" + "="*80)
    print("📊 BENCHMARK REPORT")
    print("="*80)
    
    print("\n【実行環境】")
    print(f"  Python:  {python_version}")
    print(f"  FastAPI: {fastapi_version}")
    print(f"  Pydantic: {pydantic_version}")
    
    print("\n【比較表】")
    print("-"*80)
    print(f"{'Method':<25} {'Avg (ms)':<12} {'Median (ms)':<12} {'RPS':<12} {'Stdev (ms)':<12}")
    print("-"*80)
    
    for name, result in results:
        print(f"{name:<25} {result['avg_response_time']:>10.2f}  "
              f"{result['median_response_time']:>10.2f}  "
              f"{result['rps']:>10.2f}  "
              f"{result['stdev']:>10.2f}")
    
    print("-"*80)
    
    # 比較分析
    old_way = results[0][1]
    new_way = results[1][1]
    
    avg_diff_percent = ((old_way['avg_response_time'] - new_way['avg_response_time']) 
                        / old_way['avg_response_time'] * 100)
    rps_diff_percent = ((new_way['rps'] - old_way['rps']) / old_way['rps'] * 100)
    
    print("\n【考察】")
    print("-"*80)
    
    if avg_diff_percent > 0:
        print(f"✅ 戻り値型ヒント (new-way) の方が平均 {avg_diff_percent:.2f}% 高速でした。")
        print(f"   平均レスポンス時間: {old_way['avg_response_time']:.2f}ms → {new_way['avg_response_time']:.2f}ms")
    elif avg_diff_percent < 0:
        print(f"⚠️  response_model (old-way) の方が平均 {abs(avg_diff_percent):.2f}% 高速でした。")
        print(f"   平均レスポンス時間: {old_way['avg_response_time']:.2f}ms → {new_way['avg_response_time']:.2f}ms")
    else:
        print("📊 両者のパフォーマンスはほぼ同等です。")
    
    print(f"\n   RPS (Requests Per Second):")
    print(f"   - Old Way: {old_way['rps']:.2f} req/s")
    print(f"   - New Way: {new_way['rps']:.2f} req/s")
    print(f"   - 差分: {rps_diff_percent:+.2f}%")
    
    if abs(avg_diff_percent) < 5:
        print("\n💡 結論: パフォーマンス差は微小（5%未満）であり、実用上はほぼ同等と言えます。")
        print("   型ヒントの方がPythonらしい記述ができるため、可読性の観点から推奨されます。")
    elif avg_diff_percent > 5:
        print("\n💡 結論: 戻り値型ヒントの方が明確に高速です。")
    else:
        print("\n💡 結論: response_modelの方が明確に高速です。")
    
    print("="*80)
    print()


async def main():
    """メイン処理"""
    import fastapi
    import pydantic
    
    # バージョン情報取得
    python_version = sys.version.split()[0]
    fastapi_version = fastapi.__version__
    pydantic_version = pydantic.__version__
    
    print("="*80)
    print("🚀 FastAPI Performance Benchmark")
    print("   Return Type Annotation vs response_model")
    print("="*80)
    
    base_url = "http://127.0.0.1:8000"
    requests_count = 500  # リクエスト数
    
    print(f"\n⚙️  Configuration:")
    print(f"   Server: {base_url}")
    print(f"   Requests per endpoint: {requests_count}")
    
    # ベンチマーク実行
    results = await run_benchmarks(base_url, requests_count)
    
    # レポート出力
    print_report(results, python_version, fastapi_version, pydantic_version)


if __name__ == "__main__":
    asyncio.run(main())

```
